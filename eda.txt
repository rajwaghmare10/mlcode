
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# 1. Load Dataset
df = pd.read_csv("your_dataset.csv")
print("Dataset Loaded\n")


# 2. Basic Dataset Information

print("HEAD:\n", df.head())
print("\nSHAPE:", df.shape)
print("\nINFO:")
print(df.info())
print("\nDESCRIBE:\n", df.describe())
print("\nNULL VALUES:\n", df.isnull().sum())
print("\nDUPLICATES:", df.duplicated().sum())


# 3. Univariate Analysis
# Histogram → shows distribution of a numerical column
sns.histplot(df["num_column"], kde=True)
plt.title("Histogram - Numeric Column")
plt.show()

# Countplot → shows frequency of categories in a column
sns.countplot(x=df["cat_column"])
plt.title("Count Plot - Categorical Column")
plt.show()

# 4. Bivariate Analysis
# Scatterplot → used to see the relationship between two numeric variables
sns.scatterplot(x=df["num_column"], y=df["num_column2"])
plt.title("Scatter Plot - Num vs Num")
plt.show()

# Boxplot → compares numerical values across categories
sns.boxplot(x=df["cat_column"], y=df["num_column"])
plt.title("Box Plot - Cat vs Num")
plt.show()

# 5. Correlation Heatmap
# Heatmap → shows correlation strength between numerical features
plt.figure(figsize=(10,6))
sns.heatmap(df.corr(), annot=True, cmap="coolwarm")
plt.title("Correlation Heatmap")
plt.show()

# 6. Pairplot
# Pairplot → shows pairwise relationships between all numeric columns
sns.pairplot(df)
plt.show()
