    echo "<pre>";

    // Array keys
    echo "Keys: ";
    print_r(array_keys($user));
    echo "<br><br>";

<?php
session_start();

// Hardcoded credentials
$correctUser = "admin";
$correctPass = "12345";

// Get input
$user = $_POST['username'];
$pass = $_POST['password'];

// Verify login
if ($user == $correctUser && $pass == $correctPass) {

    // Create session
    $_SESSION['username'] = $user;

    // Set cookie for 1 day if "Remember Me" checked
    if (isset($_POST['remember'])) {
        setcookie("username", $user, time() + 86400); // 24 hours
    }

    header("Location: welcome.php");
} else {
    echo "<h3>Invalid Login! Try again.</h3>";
    echo "<a href='login.php'>Go Back</a>";
}
?>
